  class SoftGraphBrain:
      """
      Constructs temporary knowledge graphs at query time
      Applies graph algorithms for sophisticated reranking

      def build_query_graph(self, query, top_k=20):
          """Build graph from top-k search results"""
          import networkx as nx

          # Get candidates
          candidates = self.search(query, limit=top_k)

          # Create temporary graph
          G = nx.DiGraph()

          # Add nodes
          for i, item in enumerate(candidates):
              G.add_node(i,
                  item=item,
                  score=item.score,
                  timestamp=item.payload['timestamp'],
                  phase=item.payload['phase']
              )

          # Add edges based on relationships
          for i, item_i in enumerate(candidates):
              # Temporal edges (older → newer)
              for j, item_j in enumerate(candidates):
                  if item_j.payload['timestamp'] > item_i.payload['timestamp']:
                      if self.semantic_similar(item_i, item_j, threshold=0.7):
                          G.add_edge(i, j, type='evolution', weight=0.8)

              # Session edges (same conversation)
              for j, item_j in enumerate(candidates):
                  if item_i.payload.get('session_id') == item_j.payload.get('session_id'):
                      G.add_edge(i, j, type='session', weight=0.9)

              # File edges (same document)
              for j, item_j in enumerate(candidates):
                  if item_i.payload['file_path'] == item_j.payload['file_path']:
                      G.add_edge(i, j, type='document', weight=0.7)

          return G, candidates

      def rerank_with_pagerank(self, query):
          """Use PageRank to identify authoritative nodes"""
          G, candidates = self.build_query_graph(query)

          # Compute PageRank
          pagerank_scores = nx.pagerank(G, weight='weight')

          # Combine with semantic scores
          final_scores = []
          for i, item in enumerate(candidates):
              semantic = item.score
              authority = pagerank_scores[i]
              final = 0.6 * semantic + 0.4 * authority  # Weighted
              final_scores.append((item, final))

          # Rerank
          return sorted(final_scores, key=lambda x: x[1], reverse=True)

      def rerank_with_centrality(self, query):
          """Use betweenness centrality to find bridge concepts"""
          G, candidates = self.build_query_graph(query)

          # Compute betweenness centrality (finds "bridge" nodes)
          centrality = nx.betweenness_centrality(G, weight='weight')

          # High centrality = connects disparate concepts
          # Good for comprehensive answers
          final_scores = []
          for i, item in enumerate(candidates):
              semantic = item.score
              bridge_value = centrality[i]
              final = 0.5 * semantic + 0.5 * bridge_value
              final_scores.append((item, final))

          return sorted(final_scores, key=lambda x: x[1], reverse=True)

      def rerank_with_community_detection(self, query):
          """Find tightly connected clusters of related knowledge"""
          G, candidates = self.build_query_graph(query)

          # Detect communities (tightly connected subgraphs)
          communities = nx.community.louvain_communities(G.to_undirected())

          # Score by community cohesion
          final_scores = []
          for i, item in enumerate(candidates):
              # Find which community this node belongs to
              community_id = None
              for idx, community in enumerate(communities):
                  if i in community:
                      community_id = idx
                      break

              # Community size = indicator of topic importance
              community_size = len(communities[community_id]) if community_id is not None
  else 1

              semantic = item.score
              community_weight = community_size / len(candidates)  # Normalized
              final = 0.7 * semantic + 0.3 * community_weight
              final_scores.append((item, final))

          return sorted(final_scores, key=lambda x: x[1], reverse=True)

      def rerank_with_shortest_path(self, query, target_concept):
          """Find shortest path to target concept"""
          G, candidates = self.build_query_graph(query)

          # Identify target node (e.g., "authentication pattern")
          target_nodes = [i for i, c in enumerate(candidates)
                         if target_concept in c.payload['content'].lower()]

          if not target_nodes:
              return candidates  # No path, return original

          target = target_nodes[0]

          # Compute shortest paths from all nodes to target
          final_scores = []
          for i, item in enumerate(candidates):
              if i == target:
                  path_length = 0
              else:
                  try:
                      path_length = nx.shortest_path_length(G, source=i, target=target)
                  except nx.NetworkXNoPath:
                      path_length = float('inf')

              semantic = item.score
              path_score = 1.0 / (1.0 + path_length)  # Shorter path = higher score
              final = 0.6 * semantic + 0.4 * path_score
              final_scores.append((item, final))

          return sorted(final_scores, key=lambda x: x[1], reverse=True)

  The Soft-Graph Advantage

  Why This Beats Precomputed Graphs

  Traditional KG (Neo4j, Microsoft GraphRAG):
  - Build entire graph upfront (hours for large corpus)
  - Store in memory/disk (GB of space)
  - Rigid structure (rebuild on document change)
  - One algorithm locked in

  AURA Soft-Graph:
  - Build graph from top-20 results only (milliseconds)
  - Zero storage (discarded after query)
  - Flexible (adapts to query context)
  - Claude chooses algorithm per query
