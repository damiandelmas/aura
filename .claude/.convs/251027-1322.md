  Purpose: Construct NetworkX graph from search results

  def build_graph(result_ids):
      """Build relationship graph from results"""
      import networkx as nx

      results = [load_result(rid) for rid in result_ids]
      G = nx.DiGraph()

      for i, r in enumerate(results):
          G.add_node(i, result=r, score=r.score)

      # Add edges based on metadata relationships
      for i, r1 in enumerate(results):
          for j, r2 in enumerate(results):
              if r1.file_path == r2.file_path:
                  G.add_edge(i, j, type='file', weight=0.8)
              if r1.session_id == r2.session_id:
                  G.add_edge(i, j, type='session', weight=0.9)

      graph_id = save_graph(G)
      return graph_id

  CLI:
  imem graph build <result-ids...>
  # Returns: graph-id for later operations

  Primitive 3: graph apply ✅ VALIDATED

  Purpose: Apply ranking algorithms to constructed graph

  def apply_algorithm(graph_id, algorithm):
      """Apply NetworkX algorithm, return ranked results"""
      import networkx as nx

      G = load_graph(graph_id)

      if algorithm == 'pagerank':
          scores = nx.pagerank(G, weight='weight')
      elif algorithm == 'centrality':
          scores = nx.betweenness_centrality(G, weight='weight')

      # Return results sorted by score
      ranked = sorted(scores.items(), key=lambda x: x[1], reverse=True)
      return [G.nodes[node_id]['result'] for node_id, score in ranked]

  CLI:
  imem graph apply <graph-id> pagerank
  imem graph apply <graph-id> centrality
  # Returns: Reranked results
