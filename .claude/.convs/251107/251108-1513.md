# Query with expansion
def query_with_expansion(query_text, primary_type):
    # Primary search
    results = search(query_text, type=primary_type)
    
    # If insufficient results, expand
    if len(results) < 5:
        related = type_relations[primary_type]
        for related_type in related:
            results += search(query_text, type=related_type)
    
    return results
