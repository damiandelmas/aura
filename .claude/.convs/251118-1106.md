  # storage/hnsw_backend.py
  import hnswlib
  from pathlib import Path

  class HNSWBackend(VectorStore):
      def __init__(self, db_path, dimension=384):
          self.db = sqlite3.connect(db_path)
          self.dimension = dimension
          self.index = hnswlib.Index(space='cosine', dim=dimension)
          self._load_or_build_index()

      def _load_or_build_index(self):
          index_path = Path('.imem/corpus.hnsw')

          if index_path.exists():
              self.index.load_index(str(index_path))
          else:
              # Build from SQLite BLOBs
              rows = self.db.execute(
                  "SELECT id, embedding FROM chunks WHERE embedding IS NOT NULL"
              ).fetchall()

              if not rows:
                  return  # No embeddings yet

              self.index.init_index(
                  max_elements=len(rows),
                  M=16,              # Connections per node
                  efConstruction=200  # Build-time quality
              )

              for row in rows:
                  chunk_id = row[0]
                  embedding = pickle.loads(row[1])
                  self.index.add_items([embedding], [chunk_id])

              self.index.save_index(str(index_path))

      def search(self, query, limit=10, filters=None, use_vector=True):
          if not use_vector:
              # Delegate to metadata search
              return self._metadata_search(query, limit, filters)

          # Get query embedding
          embedding = self.embedder.encode([query])[0]

          # HNSW search
          labels, distances = self.index.knn_query([embedding], k=limit)

          # Fetch full chunks from SQLite
          chunk_ids = labels[0]
          return self.get_by_ids(chunk_ids)
