  def reconstruct_reasoning_chain(collection_name, query):
      """
      Reconstruct complete development narrative:
      1. Find implementation
      2. Get origin conversation
      3. Get design decisions  
      4. Get evolution
      5. Get failures
      6. Get patterns
      7. Order as STORY
      """

      # Find the implementation
      primary = search(query, phase='develop')

      # Build the narrative
      narrative = {
          'conversation': get_genealogy(primary.id),
          'design': cross_phase(primary.id, 'design'),
          'evolution': get_temporal(primary.id, direction='before'),
          'failures': get_siblings(primary.id, types=['Failures']),
          'decisions': get_siblings(primary.id, types=['Decisions']),
          'implementation': get_siblings(primary.id, types=['Implementation']),
          'patterns': get_siblings(primary.id, types=['Patterns'])
      }

      return narrative
