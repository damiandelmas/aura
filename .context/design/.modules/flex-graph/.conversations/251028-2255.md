  Pattern 4: Custom Filters (Data Processing)

  Add Python functions as template filters:

  # imem/src/imem/template_filters.py

  def extract_field(content: str, field_name: str) -> str:
      """Extract structured field from markdown content"""
      import re
      # Match: **Field Name**: content
      pattern = rf'\*\*{field_name}\*\*:\s*(.+?)(?=\n\*\*|\n##|\Z)'
      match = re.search(pattern, content, re.DOTALL)
      return match.group(1).strip() if match else ""

  def truncate_smart(text: str, length: int = 200) -> str:
      """Truncate at sentence boundary"""
      if len(text) <= length:
          return text

      truncated = text[:length]
      # Find last sentence end
      last_period = max(truncated.rfind('. '), truncated.rfind('! '), truncated.rfind('? '))

      if last_period > length * 0.5:  # If we found a sentence in first 50%
          return truncated[:last_period + 1]

      return truncated + "..."

  def format_timestamp(timestamp: str) -> str:
      """Format ISO timestamp nicely"""
      from datetime import datetime
      try:
          dt = datetime.fromisoformat(timestamp.replace('Z', '+00:00'))
          return dt.strftime('%B %d, %Y at %I:%M %p')
      except:
          return timestamp

  Register filters in compose.py:
  def _render_template(results: List[Dict[str, Any]], template_name: str) -> str:
      from pathlib import Path
      from jinja2 import Environment, FileSystemLoader
      from .template_filters import extract_field, truncate_smart, format_timestamp

      template_dir = Path(__file__).parent.parent.parent / 'templates'

      env = Environment(
          loader=FileSystemLoader(str(template_dir)),
          trim_blocks=True,
          lstrip_blocks=True
      )

      # Register custom filters
      env.filters['extract_field'] = extract_field
      env.filters['truncate_smart'] = truncate_smart
      env.filters['format_timestamp'] = format_timestamp

      template = env.get_template(f"{template_name}.j2")
      return template.render(results=results)

  {# imem/templates/custom.j2 #}

      {% include "partials/failures.j2" %}
