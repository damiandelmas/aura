  1. IMEM Ingestion with LlamaIndex

  # imem/src/imem/ingest.py
  from llama_index.core.node_parser import MarkdownNodeParser

  class DocumentIngestion:
      def __init__(self):
          self.parser = MarkdownNodeParser()

      def ingest_changelog(self, file_path: Path):
          """Index changelog with H3-level chunking"""

          # Extract phase from path
          phase = self._extract_phase(file_path)  # design/designate/develop/document

          # Parse markdown
          with open(file_path) as f:
              text = f.read()

          # Extract frontmatter metadata
          frontmatter = self._extract_frontmatter(text)

          # Chunk with LlamaIndex
          nodes = self.parser.get_nodes_from_documents([{
              'text': text,
              'metadata': {'file_path': str(file_path)}
          }])

          # Index each H3 section
          for node in nodes:
              vector = self.embed(node.text)

              self.qdrant.upsert({
                  'vector': vector,
                  'payload': {
                      'source': 'changelog',
                      'phase': phase,
                      'section_type': node.metadata.get('header_path'),  # "Decisions > Use JWT"
                      'section_level': node.metadata.get('header_level'),
                      'category': frontmatter.get('type', '').split('.')[0],
                      'subtype': frontmatter.get('type', '').split('.')[1] if '.' in frontmatter.get('type', '') else
  None,
                      'timestamp': frontmatter.get('timestamp'),
                      'content': node.text
                  }
              })

      def ingest_conversation(self, markdown_path: Path, session_id: str):
          """Index conversation with H2-level chunking"""

          with open(markdown_path) as f:
              text = f.read()

          nodes = self.parser.get_nodes_from_documents([{
              'text': text,
              'metadata': {'session_id': session_id}
          }])

          for node in nodes:
              vector = self.embed(node.text)

              self.qdrant.upsert({
                  'vector': vector,
                  'payload': {
                      'source': 'conversation',
                      'session_id': session_id,
                      'section_type': node.metadata.get('header_path'),  # "User Messages"
                      'section_level': node.metadata.get('header_level'),
                      'content': node.text
                  }
              })

                help='Filter by document phase')
  @click.option('--section',
                help='Filter by section type (e.g., "Decisions", "User Messages")')
  def search(query, phase_filter, section):
      filters = {}

      if phase_filter == 'conversations':
          filters['source'] = 'conversation'
      elif phase_filter != 'all':
          filters['source'] = 'changelog'
          filters['phase'] = phase_filter

      if section:
          filters['section_type'] = section

      results = searcher.search(query, filters=filters)
