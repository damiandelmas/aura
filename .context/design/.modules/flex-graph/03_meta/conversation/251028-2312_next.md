  def _enrich_metadata(results):
      """Add temporal position, confidence signals"""
      for result in results:
          # Temporal position
          result['temporal_position'] = detect_position(
              result['timestamp'],
              result.get('temporal', [])
          )  # "current_thrust", "superseded", "failed_branch"

          # Confidence signals
          result['confidence'] = {
              'has_rationale': result['payload'].get('has_rationale'),
              'semantic_score': result['score'],
              'continuation_count': len([t for t in result.get('temporal', [])
                                        if t['timestamp'] > result['timestamp']])
          }

      return results
